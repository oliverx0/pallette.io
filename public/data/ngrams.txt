
def read_ngrams_file(filename):
  ngrams = {}
  lines = [line.strip().lower() for line in open(filename)]
  for year in range(1800, 2009):
    if year not in ngrams:
      ngrams[year] = {}
      for word in lines:
        letter = word[0]
        if letter not in ngrams[year]:
          ngrams[year][letter] = {}
        ngrams[year][letter][word] = 0
  return ngrams
